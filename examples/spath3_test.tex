    M\{##1\}\{##2\}
  }%
  \def\pgfsyssoftpath@linetotoken##1##2{%
    L\{##1\}\{##2\}
  }%
  \def\pgfsyssoftpath@curvetotoken##1##2{%
    C\{##1\}\{##2\}
  }%
  \def\pgfsyssoftpath@curvetosupportatoken##1##2{%
    Ca\{##1\}\{##2\}
  }%
  \def\pgfsyssoftpath@curvetosupportbtoken##1##2{%
    Cb\{##1\}\{##2\}
  }%
  \def\pgfsyssoftpath@closepathtoken##1##2{%
    Z\{##1\}\{##2\}
  }%
}
